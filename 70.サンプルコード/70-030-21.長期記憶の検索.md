"""
Long-term Memory からサマリーを取得するスクリプト
討論終了後、数分待ってから実行してください。
"""
import os
import argparse
from datetime import datetime
from dotenv import load_dotenv
from bedrock_agentcore.memory import MemoryClient
from bedrock_agentcore.memory.constants import StrategyType

load_dotenv()

# パネリスト定義（本体と同じ）
PANELISTS = {
    "tanaka": {"name": "田中", "role": "日本国首相"},
    "minami": {"name": "南", "role": "歴史専門家"},
    "kanemoto": {"name": "金本", "role": "経済専門家"},
    "tsuyoshi": {"name": "強", "role": "軍事専門家"},
    "taira": {"name": "平", "role": "平和主義市民"},
    "taka": {"name": "タカ", "role": "タカ派市民"},
}

def get_speaker_name(actor_id: str) -> str:
    """actor_idから表示名を取得"""
    if actor_id == "tahara":
        return "田原（司会）"
    elif actor_id in PANELISTS:
        info = PANELISTS[actor_id]
        return f"{info['name']}（{info['role']}）"
    return actor_id


def retrieve_long_term_memories(memory_id: str, session_id: str, region: str):
    """Long-term memoryからサマリーを取得"""
    
    client = MemoryClient(region_name=region)
    all_actors = ["tahara"] + list(PANELISTS.keys())
    
    print("=" * 60)
    print(f"Long-term Memory サマリー取得")
    print(f"Memory ID: {memory_id}")
    print(f"Session ID: {session_id}")
    print("=" * 60)
    
    total_memories = 0
    
    for actor_id in all_actors:
        speaker = get_speaker_name(actor_id)
        namespace = f"/strategies/summary_builtin_lodbf-2whoD9DEDz/actors/tahara/sessions/session_debate_001"
        
        try:
            memories = client.retrieve_memories(
                memory_id=memory_id,
                namespace=namespace,
                query="討論の要点と立場",
                top_k=5
            )
            
            if memories:
                print(f"\n【{speaker}】({len(memories)}件)")
                print("-" * 40)
                for i, mem in enumerate(memories, 1):
                    content = mem.get('content', '')
                    score = mem.get('score', 'N/A')
                    print(f"  {i}. {content}")
                    print(f"     (relevance: {score})")
                total_memories += len(memories)
            else:
                print(f"\n【{speaker}】記録なし")
                
        except Exception as e:
            print(f"\n【{speaker}】取得失敗: {e}")
            error_msg = str(e)
            if "memoryId" in error_msg and "pattern" in error_msg:
                print(f"\n[ERROR] Memory IDの形式が無効です")
                print(f"  指定された値: {memory_id}")
                return
            print(f"\n【{speaker}】取得失敗: {e}")            
    
    print("\n" + "=" * 60)
    print(f"合計: {total_memories}件のLong-term memory取得")
    print("=" * 60)
    
    # 抽出がまだの場合のヒント
    if total_memories == 0:
        print("\n[ヒント] Long-term memoryが0件の場合:")
        print("  - 討論終了から2-3分待ってから再実行してください")
        print("  - Memory戦略が正しく設定されているか確認してください")

def list_all_memories(region: str):
    """利用可能なMemoryリソース一覧を表示"""
    client = MemoryClient(region_name=region)
    
    print("=" * 60)
    print("利用可能なMemoryリソース一覧")
    print("=" * 60)
    
    memories = client.list_memories()
    for m in memories:
        print(f"\nName: {m.get('name', 'N/A')}")
        print(f"  ID: {m.get('id', 'N/A')}")
        print(f"  Status: {m.get('status', 'N/A')}")
        print(f"  Created: {m.get('createdAt', 'N/A')}")


def main():
    parser = argparse.ArgumentParser(description="Long-term Memoryからサマリーを取得")
    parser.add_argument("--memory-id", "-m", default=os.getenv("AGENTCORE_MEMORY_ID"))
    parser.add_argument("--session-id", "-s", default=os.getenv("DEBATE_SESSION_ID"))
    parser.add_argument("--region", "-r", default=os.getenv("AWS_DEFAULT_REGION", "us-east-1"))
    parser.add_argument("--list", "-l", action="store_true")
    
    args = parser.parse_args()
    
    if args.list:
        list_all_memories(args.region)
        return
    
    if not args.memory_id:
        print("[ERROR] Memory IDが指定されていません")
        return
    
    retrieve_long_term_memories(args.memory_id, args.session_id, args.region)

if __name__ == "__main__":
    main()
